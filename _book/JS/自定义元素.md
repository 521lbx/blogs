### 自定义元素
* 自定义元素的名字必须包含一个破折号（-）
<!--sec data-title="demo代码" data-id="section0" data-show=true ces-->
```
<!DOCTYPE html>
<html>
	<head>
		<meta charset="UTF-8">
		<title></title>
		<style>
			greeting{
				color: red;
				display: block;
			}
		</style>
	</head>
	<body>
		<div id="">
			
		</div>
		<greeting>你好，世界！</greeting>
		<greeting>你好，世界！</greeting>
		<my-element content="Custom Element">
		</my-element>
		<script>
			class MyElement extends HTMLElement {
			  get content() {
			  	console.log(222);
			    return this.getAttribute('content');
			  }
			
			  set content(val) {
			  	console.log(111, val);
			    this.setAttribute('content', val);
			  }
			}
			window.customElements.define('my-element', MyElement);
			function customTag(tagName, fn){
			  Array
			    .from(document.getElementsByTagName(tagName))
			    .forEach(fn);
			}
			
			function myElementHandler(element) {
				console.log(333, element.content);
			  element.textContent = element.content;
				console.log(444, element);
			}
			
			customTag('my-element', myElementHandler);
		</script>
	</body>
</html>
```
